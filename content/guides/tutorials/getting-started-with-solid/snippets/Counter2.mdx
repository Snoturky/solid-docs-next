```tsx
import { createSignal } from "solid-js";

function Counter() {
  const [getCount, setCount] = createSignal(0);

  const increment = () => {
    setCount(getCount() + 1);
  };

  return (
    <div>
      Current count: {getCount()}
      <button onClick={increment}>Increment</button>
    </div>
  );
}
```
