```tsx
import { createSignal } from "solid-js";

export function AddBook(props) {
  const [getInput, setInput] = createSignal("");
  const [getQuery, setQuery] = createSignal("");

  return (
    <form>
      <div>
        <label for="title">Search books</label>
        <input
          id="title"
          value={getInput()}
          onInput={(e) => {
            setInput(e.currentTarget.value);
          }}
        />
      </div>
      <button
        type="submit"
        onClick={(e) => {
          e.preventDefault();
          setQuery(getInput());
        }}
      >
        Search
      </button>
    </form>
  );
}
```
