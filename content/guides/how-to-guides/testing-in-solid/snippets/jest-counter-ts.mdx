```ts
import { createSignal, type VoidComponent } from "solid-js";

export const Counter: VoidComponent = () => {
  const [getCount, setCount] = createSignal(1);
  const increment = () => setCount(getCount() + 1);

  return (
    <button type="button" onClick={increment}>
      {getCount().toString()}
    </button>
  );
}
```
